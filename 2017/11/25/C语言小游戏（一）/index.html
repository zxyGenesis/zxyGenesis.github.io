C语言小游戏（一）

发表于 2017-11-25

猜数游戏

  游戏简介：这是一个简单的游戏：随机产生一个目标数字，比较玩家输入数值与目标数字，给出“再大（小）一点”的提示，玩家根据提示一直重复输入数值直至猜对为止。

猜数判定

  首先我们想到的是要判断输入数值与目标数字的关系，不难想到，这是通过if语句可以轻松实现的条件分支：

do{
printf(“请猜一个0~999的整数:”);

scanf("%d",&no);//玩家输入

if(no>ans)
    printf("再小一点\n");
else if(no<ans)
    printf("再大一点\n");
else 
    printf("答案正确\n");

}while(no!=ans);
//猜数判断

#include

#include //包含rand、srand函数

#include //包含time函数
int main(){
int no;//输入数值
int ans;//目标数字
int remain = 10;//设置猜数次数

srand(time(NULL));//将时间作为随机数的种子，此处的NULL表示空指针，指针可能以后介绍
ans = rand()%1000;//对rand()取模以得到0~999的整数 
printf("请猜一个0~999的整数:");
do{
    printf("还剩%d次机会：",remain);

    scanf("%d",&no);//玩家输入
    remain--;//剩余次数减1

    if(no>ans)
        printf("再小一点\n");
    else if(no<ans)
        printf("再大一点\n");

}while(no!=ans&&remain>0);
    //猜数判断    
if(no!=ans)
    printf("很遗憾，正确答案是%d。\n",ans);
else{
    printf("回答正确。\n");
    printf("你用了%d次猜对了。\n",10-remain); 
}    


return 0;

}
